📋 Requirements Checklist

  • SSH key-only & firewall rules

  • Automated DB backups (cron)

💾 Virtual Lab Environment Setup

🚀 One-Copy Setup Commands (Ubuntu/Debian)

# Install VirtualBox
sudo apt update && sudo apt install virtualbox -y

# Download Ubuntu Server ISO
wget https://releases.ubuntu.com/22.04/ubuntu-22.04-live-server-amd64.iso

# Create VM via command line (VBoxManage)
VBoxManage createvm --name "LinuxLab" --ostype Ubuntu_64 --register
VBoxManage modifyvm "LinuxLab" --memory 2048 --cpus 2
VBoxManage storagectl "LinuxLab" --name "SATA" --add sata --controller IntelAhci
VBoxManage createhd --filename "LinuxLab.vdi" --size 20000
VBoxManage storageattach "LinuxLab" --storagectl "SATA" --port 0 --device 0 --type hdd --medium "LinuxLab.vdi"
VBoxManage storageattach "LinuxLab" --storagectl "SATA" --port 1 --device 0 --type dvddrive --medium ubuntu-22.04-live-server-amd64.iso

# Start VM
VBoxManage startvm "LinuxLab"
